1, check the git configuration
	git config --list

2, require for help
	git help <verb>
	git <verb> --help
	man git-<verb>

3, difference from last modify
	git diff
	git diff -staged/-cached

4, delete file from staged area
	git rm

5, check the history of commit
	git log
	git log -p -2
	git log --stat
	git log --pretty=oneline
